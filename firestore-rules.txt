rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    function isSignedIn() {
      return request.auth != null;
    }

    function isAdminOrManager() {
      return isSignedIn() &&
        (get(/databases/$(database)/documents/admins/$(request.auth.uid)).data.role in ['admin', 'manager']);
    }
    // Admins
    match /admins/{document} {
      allow read, write: if request.auth != null;
    }

    // Site content - public readable
    match /TOURS/{document} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    match /admin_tours/{document} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    match /POSTS/{document} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    match /reviews/{document} {
      allow read: if true;
      // Anyone signed in can create/update; delete restricted to admin/manager
      allow create, update: if isSignedIn();
      allow delete: if isAdminOrManager();
    }
    
    // Post comments collection
    match /post_comments/{document} {
      allow read: if true;
      // Signed-in users can create
      allow create: if isSignedIn();
      // Only admin/manager can delete
      allow delete: if isAdminOrManager();
      // Updates generally not needed; deny by default
    }
    match /settings/{document} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    match /themes/{document} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    match /about/{document} {
      allow read: if true;
      allow write: if request.auth != null;
    }
		match /banks/{document} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    // Users/customers
    match /users/{userId} {
      allow read, write: if request.auth != null;
    }
    match /customers/{userId} {
      allow read, write: if request.auth != null;
    }

    // Bookings and affiliates (authenticated)
    match /bookings/{document} {
      allow read, write: if request.auth != null;
    }
    match /affiliates/{document} {
      allow read, write: if request.auth != null;
    }
    match /partners/{document} {
      allow read, write: if request.auth != null;
    }
    match /refund-payment/{document} {
      allow read, write: if request.auth != null;
    }
    match /affiliate-payment/{document} {
      allow read, write: if request.auth != null;
    }
    // Default - deny all
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
